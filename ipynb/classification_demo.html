def plot_metric(evals_result, metric):
    for label in evals_result:
        plt.plot(evals_result[label][metric], label=label)
    plt.legend()
    plt.title('Metric during training')
    plt.xlabel('Iterations')
    plt.ylabel(metric)   

plot_metric(evals_result, metric='auc')
