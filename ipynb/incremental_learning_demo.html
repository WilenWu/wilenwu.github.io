def progress(stats):
    """Report progress information, return a string."""
    s = "{n_train} train samples({n_train_pos} positive), " + \
        "valid's auc: {test_auc:3f} " + \
        "in {duration:.2f}s"
    s = s.format(**stats)
    return s

stats = {'n_train': 0, 'n_train_pos': 0,
         'n_test': 0, 'n_test_pos': 0,
         'auc_history': [(0, 0, 0)], 
         'duration': 0.0}

# Main loop : iterate on mini-batches of examples
cls = SGDClassifier(loss='log_loss', max_iter=100, penalty='elasticnet')
for i, train in enumerate(minibatch_iterators):
    X_train = train.drop(['id', 'label'], axis = 1)
    y_train = train['label']
    tick = time.time()
    # update estimator with examples in the current mini-batch
    cls.partial_fit(X_train, y_train, classes = [1, 0])
    
    # accumulate test accuracy stats
    stats['duration'] += time.time() - tick
    stats['n_train'] += X_train.shape[0]
    stats['n_train_pos'] += sum(y_train)
    stats['train_auc'] = roc_auc_score(y_train, cls.predict_proba(X_train)[:,1])
    stats['test_auc'] = roc_auc_score(y_test, cls.predict_proba(X_test)[:,1])
    auc_history = (stats['n_train'], stats['train_auc'], stats['test_auc'])
    stats['auc_history'].append(auc_history)
    if i % 5 == 0:
        print(progress(stats))

scores = []

gbdt = GradientBoostingClassifier(max_depth=2, warm_start=True)
for i, train in enumerate(minibatch_iterators):
    X_train = train.drop(['id', 'label'], axis = 1)
    y_train = train['label']
    gbdt.n_estimators = 1 + i
    gbdt.fit(X_train, y_train)
    # accumulate test accuracy stats
    train_auc = roc_auc_score(y_train, gbdt.predict_proba(X_train)[:,1])
    test_auc = roc_auc_score(y_test, gbdt.predict_proba(X_test)[:,1])
    scores.append((train_auc, test_auc))
    if i % 5 == 0:
        print(f"[{1 + i}] trees, test's auc: {test_auc}")
